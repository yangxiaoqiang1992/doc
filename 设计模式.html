js设计模式

1、桥接模式

将抽象与其实现隔离开来，一般适用于事件驱动,监听器回调函数

        
            function a(){

            }

用于连接公用api和私有代码

        
            /**
            *  访问私有属性
            */
            var Public=function(){
                var secret='123';
                this.getSecret=function(){
                    return secret
                }
            }
            var p =new Public();
            p.getSecret()   //123
        
    
        
            /**
            *  连接多个类
            */
            function Aclass(a){
                this.a = a
            }
            function Bclass(b){
                this.b = b
            }
            function Cclass{
                this.a =new Aclass('a');
                this.b =new Bclass('b');
            }
            console.log(new Cclass('a','b'))
